// --- ĐOẠN CODE LƯU ĐIỂM DANH HOÀN THIỆN --- async function saveAttendance(studentId, studentName, photoBase64) { // 1. Lấy thông tin phiên dạy hiện hành const sessSnap = await db.ref('current_session').once('value'); const sess = sessSnap.val(); if(!sess || !sess.date) { alert("Lỗi: Giảng viên chưa thiết lập phiên dạy!"); return; } // 2. Tạo bản ghi điểm danh (Không lấy GPS) const now = new Date(); const timeStr = now.toLocaleTimeString('vi-VN'); const dateStr = now.toLocaleDateString('vi-VN'); const attendanceData = { stt: studentId, name: studentName, time: timeStr + " " + dateStr, photo: photoBase64, lat: null, // Gỡ bỏ GPS lng: null // Gỡ bỏ GPS }; // 3. Đẩy lên Firebase try { await db.ref(`attendance/${sess.date}/${sess.session}/${studentId}`).set(attendanceData); // Hiện thông báo thành công document.getElementById('status').innerText = "Chào " + studentName + "! Điểm danh thành công."; document.getElementById('status').style.color = "#28a745"; // Tự động reset màn hình sau 3 giây để người tiếp theo quét setTimeout(() => { location.reload(); }, 3000); } catch (error) { alert("Lỗi kết nối: " + error.message); } }
